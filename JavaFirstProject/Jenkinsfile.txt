pipeline {
	
		agent any

		stages {
		
			stage ('Initialize') {
				steps {
					bat 'mvn clean install'
				}
			}
			
			stage ('Test') {
				steps {
					bat 'mvn test'
				}
			}
			
			stage ('Package') {
				steps {
					bat 'mvn package'
				}
			}

			stage ('Execute') {
				steps {
					bat 'java -cp target/JavaFirstProject-1.0-SNAPSHOT.jar com.sapient.week1'
				}
			}
			
		}
	}
